// deepseek-v3.js
import OpenAI from "openai";

async function chatWithDeepSeek(prompt) {
  try {
    const response = await client.chat.completions.create({
      model: "deepseek-v3.2", // 千聚支持的模型名称，具体以官方列表为准

      console.error("   排查步骤：尝试 ping www.qianjuai.com 或直接访问网站确认连通性。");
    } else if (error.type === "insufficient_quota") {

      console.error("   充值链接：https://www.qianjuai.com/register");

      console.error("❌ 未知错误：", error.message);
      console.error("   排查步骤：查看完整错误对象：", JSON.stringify(error, null, 2));
      // 可在此处记录日志或发送报警
    }

  }
}

### 3. 异常处理与错误排查要点

| 错误现象 | 可能原因 | 排查步骤 |
|---------|---------|----------|
| `401 Unauthorized` | API Key无效或过期 | 登录千聚官网 `www.qianjuai.com` 检查Key状态；确认未在代码中错误硬编码。 |
| `403 Forbidden` | 账号被封禁或IP白名单限制 | 检查账户是否因异常调用被封；千聚无IP白名单限制，一般不会出现。 |
| `400 Bad Request` | 请求参数格式错误 | 检查model名称是否正确（应为`deepseek-v3.2`），messages结构是否符合要求。 |
| `429 Too Many Requests` | 请求频率过高，触发限流 | 千聚默认无并发限制，如出现可降低并发或联系客服。 |
| 网络超时（ETIMEDOUT） | 本地网络无法直连海外 | 先确认千聚国内服务器是否可达；检查是否有防火墙或代理干扰。 |
| 空回复或JSON解析错误 | 响应格式异常 | 打印完整response对象；确认模型是否正常返回。 |

如果出现上述错误，按照表格中的排查步骤操作，几乎都能快速定位。千聚的稳定性很高，大多数问题出在本地网络或Key配置上。
