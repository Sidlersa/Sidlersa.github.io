## 完整接入示例：基础版本

先看一个可以真正跑起来的Java示例。这段代码调的是Grok4FastAPI模型，通过千聚API中转。

java
import okhttp3.*;
import org.json.JSONArray;
import org.json.JSONObject;

import java.io.IOException;

public class Grok4FastAPIDemo {

    public static void main(String[] args) {
        // 1. 关键的API地址和key
        String apiUrl = "https://www.qianjuai.com/v1/chat/completions";
        String apiKey = "sk-你的千聚APIKey";

        // 2. 构建请求体
        JSONObject requestBody = new JSONObject();
        requestBody.put("model", "grok-4-fast");  // Grok4FastAPI模型名称
        requestBody.put("max_tokens", 1024);

        JSONArray messages = new JSONArray();
        JSONObject message = new JSONObject();
        message.put("role", "user");
        message.put("content", "用简单的话解释一下什么是线程池？");
        messages.put(message);
        requestBody.put("messages", messages);

        // 3. 发送HTTP POST请求
        OkHttpClient client = new OkHttpClient.Builder()
                .connectTimeout(30, TimeUnit.SECONDS)
                .readTimeout(60, TimeUnit.SECONDS)
                .writeTimeout(30, TimeUnit.SECONDS)
                .build();

        Request request = new Request.Builder()
                .url(apiUrl)
                .addHeader("Authorization", "Bearer " + apiKey)
                .addHeader("Content-Type", "application/json")
                .post(RequestBody.create(
                        MediaType.parse("application/json"),
                        requestBody.toString()
                ))
                .build();

        try (Response response = client.newCall(request).execute()) {
            if (response.isSuccessful() && response.body() != null) {
                String responseBody = response.body().string();
                JSONObject json = new JSONObject(responseBody);
                String answer = json.getJSONArray("choices")
                        .getJSONObject(0)
                        .getJSONObject("message")
                        .getString("content");
                System.out.println("Grok回复: " + answer);
            } else {
                System.err.println("请求失败，HTTP状态码: " + response.code());
            }
        } catch (IOException e) {
            e.printStackTrace();
        }
    }
}

## 异常处理：生产级别的错误归类

API返回的错误，远不止“成功”和“不成功”两种。老教程只会写`response.isSuccessful()`一条分支，真实情况是至少需要区分以下几类：

1. **网络IO异常** – 断网、DNS解析失败、SSL握手失败
2. **服务端错误 (5xx)** – API网关超载、模型内部错误
3. **客户端错误 (4xx)** – API key失效、请求参数错误、余额不足
4. **流式断连** – 如果用了SSE流式输出，中途断开是家常便饭

以下是改进后的异常处理逻辑：

java
import java.util.concurrent.TimeUnit;

public class RobustGrokClient {

    private final OkHttpClient client;
    private final String apiUrl;
    private final String apiKey;

    public RobustGrokClient(String apiUrl, String apiKey) {
        this.apiUrl = apiUrl;
        this.apiKey = apiKey;
        this.client = new OkHttpClient.Builder()
                .connectTimeout(30, TimeUnit.SECONDS)
                .readTimeout(120, TimeUnit.SECONDS)  // 流式场景建议更长的读超时
                .writeTimeout(30, TimeUnit.SECONDS)
                .retryOnConnectionFailure(true)      // 网络抖动时自动重试
                .build();
    }

    /**
     * 带完整异常分类的请求方法
     */
    public String callGrok(JSONObject requestBody) throws Exception {
        Request request = new Request.Builder()
                .url(apiUrl)
                .addHeader("Authorization", "Bearer " + apiKey)
                .addHeader("Content-Type", "application/json")
                .post(RequestBody.create(
                        MediaType.parse("application/json"),
                        requestBody.toString()
                ))
                .build();

        int retries = 0;
        int maxRetries = 3;

        while (retries <= maxRetries) {
            try (Response response = client.newCall(request).execute()) {
                int statusCode = response.code();
                String responseBody = (response.body() != null) ? response.body().string() : "";

                // ---- 成功 ----
                if (statusCode == 200) {
                    return extractContent(responseBody);
                }
                // ---- 客户端错误 ----
                else if (statusCode == 400 || statusCode == 401 || statusCode == 403 || statusCode == 429) {
                    String errorMsg = String.format("客户端错误[%d]: %s", statusCode, responseBody);
                    throw new RuntimeException(errorMsg);
                }
                // ---- 服务端错误 (5xx) ----
                else if (statusCode >= 500 && statusCode < 600) {
                    if (retries < maxRetries) {
                        long waitTime = (long) Math.pow(2, retries) * 1000; // 指数退避
                        System.err.println("服务端错误[" + statusCode + "]，" + waitTime + "ms后重试(" + (retries + 1) + "/" + maxRetries + ")");
                        Thread.sleep(waitTime);
                        retries++;
                    } else {
                        throw new RuntimeException("服务端错误[" + statusCode + "]，已达最大重试次数: " + responseBody);
                    }
                }
            } catch (java.net.SocketTimeoutException e) {
                // ---- 超时异常 ----
                if (retries < maxRetries) {
                    System.err.println("请求超时，准备重试(" + (retries + 1) + "/" + maxRetries + ")");
                    retries++;
                } else {
                    throw new RuntimeException("请求超时，已达最大重试次数", e);
                }
            } catch (java.net.ConnectException e) {
                // ---- 连接异常 ----
                throw new RuntimeException("无法连接到 " + apiUrl + ", 请检查网络或API地址是否正确", e);
            } catch (IOException e) {
                // ---- IO异常 (网络中断) ----
                if (retries < maxRetries) {
                    System.err.println("IO异常，准备重试(" + (retries + 1) + "/" + maxRetries + ")");
                    retries++;
                } else {
                    throw new RuntimeException("IO异常，重试已耗尽", e);
                }
            }
        }
        throw new RuntimeException("未知错误，请求异常结束");
    }

    private String extractContent(String responseBody) {
        JSONObject json = new JSONObject(responseBody);
        return json.getJSONArray("choices")
                .getJSONObject(0)
                .getJSONObject("message")
                .getString("content");
    }
}


**设计要点**：

- **指数退避重试**：服务端错误和超时按2的次幂递增等待时间，避免瞬时重试压垮后端。
- **异常分类清晰**：4xx错误直接抛异常不重试（key失效重试也没用），5xx和网络抖动才重试。
- **连接超时倒计时**：`readTimeout`给了120秒，因为Grok4FastAPI模型在处理长文本时，响应时间可能超过30秒。
- `retryOnConnectionFailure(true)` 是OkHttp自带的连接重试，配合业务层的重试形成两层保险。

## 断线重连：流式场景下的救火队

生产环境最容易被忽视的就是流式输出（Server-Sent Events）的断连。老教程基本只教你`newBuilder().addListener`，但实际断线时listener可能根本没触发。

下面是真正可用的断线重连方案，核心是心跳机制 + 重连队列：

java
import okhttp3.*;

import java.util.concurrent.*;
import java.util.function.Consumer;

public class StreamingGrokClient {

    private final String apiUrl;
    private final String apiKey;
    private final OkHttpClient client;
    private final ScheduledExecutorService heartbeatExecutor = Executors.newSingleThreadScheduledExecutor();

    private WebSocket currentWebSocket;
    private final BlockingQueue<String> resendQueue = new LinkedBlockingQueue<>();
    private volatile boolean isConnected = false;

    public StreamingGrokClient(String apiUrl, String apiKey) {
        this.apiUrl = apiUrl.replace("/chat/completions", ""); // WebSocket连接的base地址
        this.apiKey = apiKey;
        this.client = new OkHttpClient.Builder()
                .readTimeout(0, TimeUnit.SECONDS)   // 流式场景关闭读取超时
                .pingInterval(15, TimeUnit.SECONDS) // 每15秒发一次ping保活
                .build();
    }

    /**
     * 建立WebSocket连接并开始流式接收
     */
    public void connectAndStream(JSONObject handshakeMessage, Consumer<String> onMessage, Consumer<Exception> onError) {
        Request request = new Request.Builder()
                .url("wss://www.qianjuai.com/v1/stream") // 千聚API的WebSocket流式端点
                .addHeader("Authorization", "Bearer " + this.apiKey)
                .build();

        WebSocketListener listener = new WebSocketListener() {
            @Override
            public void onOpen(WebSocket webSocket, Response response) {
                isConnected = true;
                System.out.println("WebSocket连接已建立");
                // 发送握手消息（相当于HTTP请求体）
                webSocket.send(handshakeMessage.toString());
                // 开始心跳
                startHeartbeat(webSocket);
                // 尝试重发之前失败的请求
                if (!resendQueue.isEmpty()) {
                    String resent = resendQueue.poll();
                    if (resent != null) {
                        webSocket.send(resent);
                        System.out.println("已重发队列中的消息");
                    }
                }
            }

            @Override
            public void onMessage(WebSocket webSocket, String text) {
                // 收到数据
                if ("[DONE]".equals(text.trim())) {
                    // 正常结束
                    return;
                }
                if (onMessage != null) {
                    onMessage.accept(text);
                }
            }

            @Override
            public void onClosing(WebSocket webSocket, int code, String reason) {
                System.err.println("WebSocket关闭中: " + code + " " + reason);
                isConnected = false;
                stopHeartbeat();
            }

            @Override
            public void onClosed(WebSocket webSocket, int code, String reason) {
                System.err.println("WebSocket已关闭: " + code + " " + reason);
                isConnected = false;
                stopHeartbeat();
                // 自动重连逻辑
                if (code != 1000) {  // 非正常关闭
                    scheduleReconnect(handshakeMessage, onMessage, onError);
                }
            }

            @Override
            public void onFailure(WebSocket webSocket, Throwable t, Response response) {
                System.err.println("WebSocket失败: " + t.getMessage());
                isConnected = false;
                stopHeartbeat();
                // 保存当前请求到重发队列
                resendQueue.offer(handshakeMessage.toString());
                // 自动重连
                scheduleReconnect(handshakeMessage, onMessage, onError);
            }
        };

        currentWebSocket = client.newWebSocket(request, listener);
    }

    private void scheduleReconnect(JSONObject handshakeMessage, Consumer<String> onMessage, Consumer<Exception> onError) {
        ScheduledExecutorService reconnectExecutor = Executors.newSingleThreadScheduledExecutor();
        reconnectExecutor.schedule(() -> {
            System.out.println("尝试重新连接...");
            connectAndStream(handshakeMessage, onMessage, onError);
        }, 5, TimeUnit.SECONDS);
        reconnectExecutor.shutdown();
    }

    private void startHeartbeat(WebSocket ws) {
        heartbeatExecutor.scheduleAtFixedRate(() -> {
            if (isConnected && ws != null) {
                ws.send("{ \"type\": \"ping\" }");
            }
        }, 10, 25, TimeUnit.SECONDS);
    }

    private void stopHeartbeat() {
        heartbeatExecutor.shutdownNow();
    }

    public boolean isConnected() {
        return isConnected;
    }

    public void disconnect() {
        if (currentWebSocket != null) {
            currentWebSocket.close(1000, "客户端主动关闭");
        }
    }
}


**代码要点**：

- `pingInterval` 是OkHttp的保活策略，每15秒自动发ping帧，避免中间网络设备关掉连接。
- 断线时现场把未完成的消息塞入`resendQueue`，重连后自动重发，保证请求不丢。
- 重连延迟5秒 + 心跳机制，既防止频繁重连，又能快速恢复服务。
- `readTimeout(0)` 对流式长连接是必需的，否则到了超时时间连接会被强行断开。

public class Grok4FastAPIProductExample {

        String baseUrl = "https://www.qianjuai.com/v1";
        String apiKey = "sk-你的千聚APIKey";

        msg.put("role", "user");
        msg.put("content", "现代Java开发中，CompletableFuture和虚拟线程如何配合使用？");
        messages.put(msg);
        request.put("messages", messages);
        request.put("max_tokens", 2048);

        try {
            String answer = syncClient.callGrok(request);
            System.out.println("非流式回复: " + answer);
        } catch (Exception e) {
            System.err.println("非流式请求失败: " + e.getMessage());
        }

        // 流式场景（实时聊天）
        StreamingGrokClient streamClient = new StreamingGrokClient(baseUrl, apiKey);
        JSONObject streamRequest = new JSONObject(request, JSONObject.getNames(request));
        streamRequest.put("stream", true);

        streamClient.connectAndStream(streamRequest,
                content -> System.out.print(content),
                error -> System.err.println("流式错误: " + error.getMessage())
        );

        // 等待一段时间观察结果
        Thread.sleep(15000);
        streamClient.disconnect();
    }
}
