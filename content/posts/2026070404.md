我带大家写一段完整的Node.js代码。这段代码的核心作用是把一张本地图片（比如你桌面上的一个“product.jpg”）发送给Qwen-VL模型，让模型来描述这张图里有什么。

### 第二步：编写调用代码

创建一个新的文件，比如叫 `vl_test.js`，然后粘贴下面的完整代码：

javascript
// 引入 openai 库
const OpenAI = require('openai');
const fs = require('fs');
const path = require('path');

// 配置千聚平台的客户端
const client = new OpenAI({
    apiKey: 'YOUR_API_KEY_HERE', // 替换成你在千聚平台获取的 Key
    baseURL: 'https://www.qianjuai.com/v1', // 这是关键！指向千聚的 API 接口
});

// 获取本地图片的 Base64 编码
function encodeImage(imagePath) {
    const absolutePath = path.resolve(imagePath);
    console.log('正在读取图片：', absolutePath);
    const imageBuffer = fs.readFileSync(absolutePath);
    const base64Image = imageBuffer.toString('base64');
    // Qwen-VL 要求传入 data:image/jpeg;base64, 格式
    return `data:image/jpeg;base64,${base64Image}`;
}

async function main() {
    // 图片文件路径，请修改为你的本地图片路径
    const imagePath = './product.jpg'; // 建议使用 jpg 或 png 格式

    // 生成图片的 data URI
    const imageUrl = encodeImage(imagePath);

    try {
        console.log('开始请求 Qwen-VL 模型...');
        const response = await client.chat.completions.create({
            model: 'qwen-vl-plus', // 指定 Qwen-VL 模型名称（plus 版本支持更好的视觉理解）
            messages: [
                {
                    role: 'user',
                    content: [
                        {
                            type: 'text',
                            text: '请描述这张图片中有什么，并列出它的几个主要颜色。',
                        },
                        {
                            type: 'image_url',
                            image_url: {
                                url: imageUrl,
                                // 可以不用指定 detail 参数，Qwen-VL 会自动处理
                            },
                        },
                    ],
                },
            ],
            max_tokens: 500, // 限制回答长度，防止输出超长
        });

        console.log('模型回复：');
        console.log(response.choices[0]?.message?.content || '模型没有回复内容。');

    } catch (error) {
        // 处理各种常见报错
        if (error.status === 401) {
            console.error('❌ 错误 401: API 密钥无效。请检查你的 apiKey 是否正确。');
        } else if (error.status === 400) {
            console.error('❌ 错误 400: 请求格式错误。请检查图片格式是否为支持的 jpg/png，或者模型名称是否正确。');
            // 通常返回的 error.message 里有更详细的提示
            console.error('详细错误：', error.message);
        } else if (error.status === 429) {
            console.error('❌ 错误 429: 请求过于频繁。请稍后重试，或检查你的账户余额是否充足。');
        } else if (error.code === 'ENOENT') {
            console.error('❌ 错误: 图片文件未找到！请确保 imagePath 指向的文件存在。');
        } else {
            console.error('❌ 未知错误：', error);
        }
    }

}

main();

## 核心报错排查：代码跑不通怎么办？

我把我自己碰到的几个错误和解决方法写在下面，你大概率也会遇到：

### 1. “401 Invalid Authentication”
**原因**：apiKey 填错了，或者这个Key不是在千聚平台生成的（比如直接填了阿里云的Key）。

**解决方法**：去千聚平台重新复制Key，粘贴时注意不要有多余的空格。并且 **只使用千聚平台提供的 Key**。

### 2. “400 Invalid image_format”
**原因**：你的图片超出了Qwen-VL支持的类型。Qwen-VL支持 JPEG 和 PNG，像 WebP、GIF 或 BMP 是不被支持的，否则报错。

**解决方法**：把图片转换成 JPEG 或 PNG 格式。最简单的办法：用系统自带的画图工具另存为一下就行。如果你传递的是URL，也要确保URL指向的是图像文件，并且服务器没有限制访问。

### 3. “429 Too Many Requests” 或者 “Insufficient Quota”
**原因**：两种情况：一是短时间内发了太多请求；更常见的是**千聚平台账户余额不足**，你需要去充值。

**解决方法**：检查你的千聚账户余额。新用户有 $0.2 额度，如果够用但依然报429，可以休息一分钟再试。如果不够，就去 [https://www.qianjuai.com/register](https://www.qianjuai.com/register) 充值，最低1元起充。

### 4. `base64` 转换后报错：“Cannot convert undefined or null to object”
**原因**：最常见的原因是 `fs.readFileSync(imagePath)` 读取图片失败，导致 `base64Image` 是 `undefined`。

**解决方法**：在你填写的 `imagePath` 前加一个 `console.log` 打印它的绝对路径，看看这个路径下到底有没有图片。比如我的图片放在桌面上，就直接用绝对路径代替：`/Users/myname/Desktop/product.jpg`。
