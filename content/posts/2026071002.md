#### 4. Function Calling（工具调用）

如果你想把 AI 的能力嵌入到业务流程中，比如让 AI 根据用户的指令直接查天气、调用内部 API，Function Calling 是必须的。智谱 GLM-4 对 Function Calling 支持得非常好。

python
from openai import OpenAI

client = OpenAI(
    api_key="sk-xxxx",
    base_url="https://www.qianjuai.com/v1"
)

# 定义工具函数（模拟一个天气查询工具）
def get_current_weather(location, unit="celsius"):
    """获取指定地点的天气"""
    # 这是模拟数据，真实场景需要调用外部 API
    return f"{location} 当前温度是 22 摄氏度，天气晴朗。"

# 定义 Tool
tools = [
    {
        "type": "function",
        "function": {
            "name": "get_current_weather",
            "description": "获取指定地点的当前天气",
            "parameters": {
                "type": "object",
                "properties": {
                    "location": {
                        "type": "string",
                        "description": "城市名称，例如：北京"
                    },
                    "unit": {
                        "type": "string",
                        "enum": ["celsius", "fahrenheit"]
                    }
                },
                "required": ["location"]
            }
        }
    }
]

# 第一次调用：让 AI 决定是否调用工具
response = client.chat.completions.create(
    model="glm-4",
    messages=[
        {"role": "user", "content": "北京今天天气怎么样？"}
    ],
    tools=tools,
    tool_choice="auto"
)

response_message = response.choices[0].message
tool_calls = response_message.tool_calls

# 如果 AI 决定要调用工具

    # 解析工具调用

        "get_current_weather": get_current_weather,
    }
