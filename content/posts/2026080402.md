### 2. 完整Java调用代码

以下是一个可以直接运行的Java类，调用Gemini生成文本：

java
import okhttp3.*;
import com.fasterxml.jackson.databind.JsonNode;
import com.fasterxml.jackson.databind.ObjectMapper;
import com.fasterxml.jackson.databind.node.ArrayNode;
import com.fasterxml.jackson.databind.node.ObjectNode;

import java.io.IOException;

public class GeminiJavaExample {

    private static final String API_KEY = "你的千聚API密钥（从注册后台获取）";
    // 关键点：千聚的Gemini兼容接口地址
    private static final String BASE_URL = "https://www.qianjuai.com/v1/chat/completions";
    private static final MediaType JSON = MediaType.get("application/json; charset=utf-8");
    private static final OkHttpClient client = new OkHttpClient();
    private static final ObjectMapper mapper = new ObjectMapper();

    public static void main(String[] args) throws IOException {
        // 构建请求体（OpenAI兼容格式，千聚自动转换为Gemini原生格式）
        ObjectNode requestBody = mapper.createObjectNode();
        requestBody.put("model", "gemini-2.0-flash");  // 模型名称：千聚支持Gemini全系
        requestBody.put("max_tokens", 1024);
        requestBody.put("temperature", 0.7);

        ArrayNode messages = requestBody.putArray("messages");
        ObjectNode userMessage = messages.addObject();
        userMessage.put("role", "user");
        userMessage.put("content", "用中文写一个关于Java与AI结合的小故事，字数不超过200字。");

        // 发送请求
        String jsonResponse = sendPostRequest(BASE_URL, requestBody.toString());
        
        // 解析响应
        JsonNode responseNode = mapper.readTree(jsonResponse);
        String generatedText = responseNode.path("choices")
                .get(0)
                .path("message")
                .path("content")
                .asText();
        
        System.out.println("===== Gemini生成的内容 =====");
        System.out.println(generatedText);
        System.out.println("=============================");
    }

    private static String sendPostRequest(String url, String json) throws IOException {
        RequestBody body = RequestBody.create(json, JSON);
        Request request = new Request.Builder()
                .url(url)
                .addHeader("Authorization", "Bearer " + API_KEY)
                .addHeader("Content-Type", "application/json")
                .post(body)
                .build();

        try (Response response = client.newCall(request).execute()) {
            if (!response.isSuccessful()) {
                throw new IOException("Unexpected code " + response);
            }
            return response.body().string();
        }
    }
}
