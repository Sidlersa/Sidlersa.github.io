# ---------- 配置区域结束 ----------

# 构造请求头（固定格式，不要改动）

    # 检查请求是否成功
    if response.status_code == 200:
        # 解析返回的JSON数据
        result = response.json()
        # 提取并打印AI的回复内容
        ai_reply = result["choices"][0]["message"]["content"]
        print("o4-mini 回复如下：")
        print(ai_reply)
    else:

except Exception as e:
