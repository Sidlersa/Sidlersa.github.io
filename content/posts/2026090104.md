export async function createCompletion(taskType, messages) {
  const config = modelsConfig[taskType] || modelsConfig.default;
  const client = new OpenAI(config);
  
  const response = await client.chat.completions.create({
    model: config.model,
    messages,
    stream: true

  }
}
