const client = new OpenAI({

const tasks = ['手机', '笔记本电脑', '耳机', '智能手表', '平板'];

async function generateAll() {

### 写法6：带错误重试和限流的稳健写法

javascript
const OpenAI = require('openai');

const client = new OpenAI({
    baseURL: 'https://www.qianjuai.com/v1',
    apiKey: '你的千聚API密钥',
    maxRetries: 3,
});

async function callWithRetry(task) {
    for (let attempt = 0; attempt < 3; attempt++) {
        try {
            const response = await client.chat.completions.create({
                model: 'o4-mini',
                messages: [{ role: 'user', content: task }],
            });
            return response.choices[0].message.content;
        } catch (error) {
            console.log(`第${attempt+1}次尝试失败，原因：${error.message}`);
            await new Promise(resolve => setTimeout(resolve, 1000 * attempt));
        }
    }
    throw new Error('三次重试后仍失败');
}

const result = await callWithRetry('请写一段300字的商品介绍文案');


- **Token耗量**：正常情况约420个；失败重试时会重复消耗Token
- **费用**：正常约0.0042元；重试两次约0.0126元
- **优点**：健壮性好，适合生产环境
- **缺点**：重试机制会额外增加Token消耗，需要合理配置最大重试次数

## 费用差异深扒：为什么差这么多？

从上面的对比可以看出，**写法2（非流式全量输出）** 比**写法1（基础流式调用）** 多消耗了近一倍的Token，这多出来的部分不是模型真的生成了更多内容，而是模型在等待和填充时的浪费。

更有意思的是，**写法3（带历史上下文的轮询）** 如果使用不当，Token消耗会指数级增长。每次调用都把整个历史对话传进去——那些之前生成的文案描述，根本和新商品不相干，但还是要花钱。

反观**写法4（结构化输出）** 和**写法5（异步并发调用）**，在于它们用最少的Token完成了最多的有效工作。结构化输出让模型用最精炼的方式给出结果；异步并发用时间换空间，充分利用千聚平台的并发能力。

**核心结论**：在多轮对话场景，一定要手动清理历史记录（比如只保留最近3轮）；在单一任务场景，用流式+结构化输出是最优解；在批量生产场景，用异步并发+流式，效果惊人。
