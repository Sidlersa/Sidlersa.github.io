## 省钱代码：实战示例（含上下文控制）

为了进一步降低token消耗成本，我们可以在代码中加入**智能上下文管理**：

python
import openai
from collections import deque

openai.base_url = "https://www.qianjuai.com/v1"
openai.api_key = "sk-你的千聚API Key"

# 一个简单的滑动窗口上下文管理器
class SmartContext:
    def __init__(self, max_tokens=32000):
        self.messages = deque()
        self.total_tokens = 0
        self.max_tokens = max_tokens

    def add_message(self, role, content, token_count):
        while self.total_tokens + token_count > self.max_tokens and self.messages:
            removed = self.messages.popleft()
            self.total_tokens -= removed.get("tokens", 0)
        new_msg = {"role": role, "content": content, "tokens": token_count}
        self.messages.append(new_msg)
        self.total_tokens += token_count

    def get_context(self):
        return [{"role": m["role"], "content": m["content"]} for m in self.messages]

# 使用示例
ctx = SmartContext(max_tokens=48000)
ctx.add_message("user", "帮我分析这份100页的研发报告", 20000)
ctx.add_message("assistant", "初步分析完成，共发现3个关键问题...", 5000)

# 然后调用DeepSeek V3.2
response = openai.chat.completions.create(
    model="deepseek-chat",
    messages=ctx.get_context(),
    stream=True
)

for chunk in response:
    if chunk.choices[0].delta.content:
        print(chunk.choices[0].delta.content, end="")
