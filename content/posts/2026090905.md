### 3. 流式输出（Streaming），更快响应

很多场景下，比如构建聊天机器人，或者需要实时反馈的应用，推荐使用流式输出。这样用户可以一边接收一边阅读，体验更好。

javascript
async function streamMoonshotAPI(model, messages) {
  try {
    const stream = await client.chat.completions.create({
      model: model,
      messages: messages,
      stream: true,
    });

    for await (const chunk of stream) {
      console.log(chunk.choices[0]?.delta?.content || '');
    }
  } catch (error) {
    console.error('流式调用失败:', error);
    throw error;
  }
}

// 使用示例

  { role: 'user', content: '帮我写一个简单的Promise示例' },

稳定调用的关键，在于优雅的错误处理和重试。MoonshotAPI 偶尔可能遇到瞬时错误（比如 429 限流或 500 服务端错误）。你可以用 `retry` 库来增强调用稳定性。

javascript
const retry = require('retry');

async function callWithRetry(model, messages, maxRetries = 3) {
  const operation = retry.operation({
    retries: maxRetries,
    factor: 2,
    minTimeout: 1000,
    maxTimeout: 10000,
  });

  return new Promise((resolve, reject) => {
    operation.attempt(async (currentAttempt) => {
      try {
        const result = await callMoonshotAPI(model, messages);
        resolve(result);
      } catch (err) {
        console.log(`第 ${currentAttempt} 次尝试失败，重试中...`);
        if (operation.retry(err)) {
          return;
        }
        reject(err);
      }
    });
  });
}

// 使用示例
const result = await callWithRetry('gpt-4o', [
  { role: 'user', content: '用500字解释相对论' },
]);
console.log(result);
