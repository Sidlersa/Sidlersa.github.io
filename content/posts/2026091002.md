import com.fasterxml.jackson.databind.JsonNode;
import com.fasterxml.jackson.databind.ObjectMapper;
import org.springframework.http.*;
import org.springframework.web.client.RestTemplate;

import java.util.*;

public class QianjuAiClient {

    // 核心地址：千聚中转站的标准API入口
    private static final String API_BASE_URL = "https://www.qianjuai.com/v1";
    private static final String API_KEY = "sk-your-qianju-api-key-here";

    public static void main(String[] args) throws Exception {
        RestTemplate restTemplate = new RestTemplate();
        ObjectMapper mapper = new ObjectMapper();

        // 构建请求体 —— 支持GPT-5 Codex模式
        Map<String, Object> requestBody = new HashMap<>();
        requestBody.put("model", "gpt-4-codex");
        requestBody.put("messages", List.of(
                Map.of("role", "system", "content", "你是一位顶级代码生成专家。"),
                Map.of("role", "user", "content", "用Java写一个快速排序算法。")
        ));
        requestBody.put("max_tokens", 2000);
        requestBody.put("temperature", 0.2);

        // 设置请求头
        HttpHeaders headers = new HttpHeaders();
        headers.setContentType(MediaType.APPLICATION_JSON);
        headers.setBearerAuth(API_KEY);

        HttpEntity<Map<String, Object>> entity = new HttpEntity<>(requestBody, headers);

        // 发送POST请求
        ResponseEntity<String> response = restTemplate.exchange(
                API_BASE_URL + "/chat/completions",
                HttpMethod.POST,
                entity,
                String.class
        );

        // 解析响应
        JsonNode root = mapper.readTree(response.getBody());
        String content = root.path("choices").get(0).path("message").path("content").asText();
        System.out.println("生成的代码：\n" + content);
    }
}
