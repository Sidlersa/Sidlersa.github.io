import okhttp3.*;
import com.fasterxml.jackson.databind.ObjectMapper;
import com.fasterxml.jackson.databind.node.ArrayNode;
import com.fasterxml.jackson.databind.node.ObjectNode;
import java.io.IOException;

public class ClaudeSonnetClient {
    // 关键点：替换为千聚AI中转站的API地址
    private static final String API_URL = "https://www.qianjuai.com/v1/chat/completions";
    // 关键点：替换为你从千聚获得的API Key
    private static final String API_KEY = "你的千聚API_Key";

    private final OkHttpClient client;
    private final ObjectMapper mapper;

    public ClaudeSonnetClient() {
        this.client = new OkHttpClient.Builder()
                .connectTimeout(60, java.util.concurrent.TimeUnit.SECONDS)
                .readTimeout(120, java.util.concurrent.TimeUnit.SECONDS)
                .build();
        this.mapper = new ObjectMapper();
    }

    public String callClaudeSonnet(String userMessage) throws IOException {
        // 构建请求体
        ObjectNode requestBody = mapper.createObjectNode();
        requestBody.put("model", "claude-sonnet-4-20250224"); // 指定模型
        requestBody.put("max_tokens", 4096);

        // 构建消息列表
        ArrayNode messages = mapper.createArrayNode();
        ObjectNode systemMessage = mapper.createObjectNode();
        systemMessage.put("role", "system");
        systemMessage.put("content", "你是一位资深的Java开发工程师，请用专业、清晰的代码回答我的问题。");
        messages.add(systemMessage);

        ObjectNode userMessageNode = mapper.createObjectNode();
        userMessageNode.put("role", "user");
        userMessageNode.put("content", userMessage);
        messages.add(userMessageNode);

        requestBody.set("messages", messages);

        // 构建HTTP请求
        Request request = new Request.Builder()
                .url(API_URL)
                .addHeader("Authorization", "Bearer " + API_KEY)
                .addHeader("Content-Type", "application/json")
                .post(RequestBody.create(
                        MediaType.parse("application/json; charset=utf-8"),
                        mapper.writeValueAsString(requestBody)
                ))
                .build();

        // 发送请求并获取响应
        try (Response response = client.newCall(request).execute()) {
            if (!response.isSuccessful()) {
                throw new IOException("请求失败，状态码: " + response.code() + "，响应: " + response.body().string());
            }
            String responseBody = response.body().string();
            // 解析JSON并提取AI回复内容
            ObjectNode jsonResponse = mapper.readValue(responseBody, ObjectNode.class);
            return jsonResponse.get("choices").get(0).get("message").get("content").asText();
        }
    }

    // 一个简单的main方法用于测试
    public static void main(String[] args) {
        ClaudeSonnetClient client = new ClaudeSonnetClient();
        try {
            String question = "请用Java写一个高效的LRU缓存算法，并解释其工作原理。";
            String answer = client.callClaudeSonnet(question);
            System.out.println("Claude Sonnet的回答：");
            System.out.println(answer);
        } catch (IOException e) {
            e.printStackTrace();
        }
    }
}

## 实战代码：Java调用Claude Sonnet的10种常见场景

这里再给你一个更完整的示例，展示如何用Java流式调用Claude Sonnet——这样可以一边生成代码一边实时显示，体验更好：

java
import okhttp3.*;
import com.fasterxml.jackson.databind.ObjectMapper;
import com.fasterxml.jackson.databind.node.*;
import java.io.*;

public class StreamClaudeSonnet {
    private static final String API_URL = "https://www.qianjuai.com/v1/chat/completions";
    private static final String API_KEY = "你的千聚API_Key";

    public static void main(String[] args) throws IOException {
        OkHttpClient client = new OkHttpClient.Builder()
                .connectTimeout(60, java.util.concurrent.TimeUnit.SECONDS)
                .readTimeout(300, java.util.concurrent.TimeUnit.SECONDS)
                .build();
        ObjectMapper mapper = new ObjectMapper();

        // 构建流式请求
        ObjectNode requestBody = mapper.createObjectNode();
        requestBody.put("model", "claude-sonnet-4-20250224");
        requestBody.put("stream", true);  // 开启流式输出
        requestBody.put("max_tokens", 8192);

        // 构建消息
        ArrayNode messages = mapper.createArrayNode();
        ObjectNode userMessage = mapper.createObjectNode();
        userMessage.put("role", "user");
        userMessage.put("content", "用Java实现一个完整的RESTful API服务，包括用户注册、登录和JWT认证。");
        messages.add(userMessage);
        requestBody.set("messages", messages);

        Request request = new Request.Builder()
                .url(API_URL)
                .addHeader("Authorization", "Bearer " + API_KEY)
                .post(RequestBody.create(
                        MediaType.parse("application/json"),
                        mapper.writeValueAsString(requestBody)
                ))
                .build();

        try (Response response = client.newCall(request).execute()) {
            if (!response.isSuccessful()) {
                System.err.println("请求失败: " + response.code());
                return;
            }
            ResponseBody body = response.body();
            if (body == null) return;
            
            BufferedReader reader = new BufferedReader(new InputStreamReader(body.byteStream()));
            String line;
            while ((line = reader.readLine()) != null) {
                if (line.startsWith("data: ") && !line.equals("data: [DONE]")) {
                    String json = line.substring(6);
                    ObjectNode chunk = mapper.readValue(json, ObjectNode.class);
                    String content = chunk.get("choices")
                            .get(0)
                            .get("delta")
                            .path("content")
                            .asText("");
                    System.out.print(content);  // 实时打印
                }
            }
        }
    }
}


**为什么推荐流式**：
- **响应更快**：不用等全部生成，看到第一个Token就开始显示。
- **用户体验好**：在网页或控制台应用中，可以做成“打字机”效果，而不是白屏等待。
- **超时更友好**：Claude Sonnet处理超长代码生成时，流式可以有效避免连接超时。
