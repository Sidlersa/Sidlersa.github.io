## 第三坑：你的Node.js代码写得不对，钱被白白烧掉了

这是最容易被忽视的坑：**代码层面**。

**错误写法：反复建立新的连接**

javascript
// 错误：每次调用都新建一个openai客户端

async function askDeepSeek(prompt) {
    const openai = new OpenAI({
        baseURL: "https://www.qianjuai.com/v1",
        apiKey: process.env.QIANJU_API_KEY
    });

    const completion = await openai.chat.completions.create({
        model: "deepseek-chat",

这样写，每一次函数调用都会创建一个新的HTTP连接，既消耗服务器资源，又容易被限流。而且每次请求都会重新设置上下文，导致AI模型没法记住之前的对话，Token就白白浪费在重复说明上。

**正确写法：复用连接 + 管理上下文**

javascript
// 正确：单例模式复用客户端
const OpenAI = require('openai');

const openai = new OpenAI({
    baseURL: "https://www.qianjuai.com/v1",
    apiKey: process.env.QIANJU_API_KEY
});

// 维护一个上下文数组
let messageHistory = [];

async function chat(prompt) {
    messageHistory.push({ role: "user", content: prompt });
    
    const completion = await openai.chat.completions.create({
        model: "deepseek-chat",
        messages: messageHistory // 带上历史上下文
    });
    
    const reply = completion.choices[0].message.content;
    messageHistory.push({ role: "assistant", content: reply });
    
    return reply;
}
