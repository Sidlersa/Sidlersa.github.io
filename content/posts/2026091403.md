### 第三步：流式输出（更高级的用法）

如果你需要像 ChatGPT 那样一个字一个字往外蹦，用流式输出：

java
import cn.hutool.http.HttpRequest;
import cn.hutool.http.HttpResponse;
import cn.hutool.json.JSONObject;

import java.io.BufferedReader;
import java.io.InputStreamReader;

public class QianJuAIStreamClient {
    private static final String API_KEY = "YOUR_QIANJU_API_KEY";
    private static final String BASE_URL = "https://www.qianjuai.com/v1";

    public static void main(String[] args) {
        String model = "doubao-pro-32k-0325";
        JSONObject requestBody = new JSONObject();
        requestBody.set("model", model);
        requestBody.set("messages", new Object[]{
            new JSONObject().set("role", "user").set("content", "请用中文讲解一下 Java 中的 Stream API。")
        });
        requestBody.set("stream", true); // 开启流式输出

        // 发起流式请求
        HttpResponse response = HttpRequest.post(BASE_URL + "/chat/completions")
            .header("Authorization", "Bearer " + API_KEY)
            .header("Content-Type", "application/json")
            .body(requestBody.toString())
            .executeAsync();

        if (response.getStatus() == 200) {
            try (BufferedReader reader = new BufferedReader(
                    new InputStreamReader(response.bodyStream(), "UTF-8"))) {
                String line;
                while ((line = reader.readLine()) != null) {
                    if (line.startsWith("data: ")) {
                        String data = line.substring(6);
                        if (!"[DONE]".equals(data)) {
                            JSONObject json = new JSONObject(data);
                            String content = json.getJSONArray("choices")
                                    .getJSONObject(0)
                                    .getJSONObject("delta")
                                    .getStr("content", "");
                            System.out.print(content);
                        }
                    }
                }
            } catch (Exception e) {
                e.printStackTrace();
            }
        } else {
            System.out.println("请求失败，状态码：" + response.getStatus());
        }
    }
}
