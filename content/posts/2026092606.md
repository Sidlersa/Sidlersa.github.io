## Java实战：一个完整的省钱调用示例

下面我贴一个标准的OkHttp调用示例，你直接复制就能用。关键就是那一行`baseUrl`：

java
import okhttp3.*;
import com.alibaba.fastjson.JSON;
import com.alibaba.fastjson.JSONObject;

public class QiangJuAPIDemo {
    
    public static void main(String[] args) {
        // 关键行：使用千聚API中转站的地址
        String baseUrl = "https://www.qianjuai.com/v1";
        // 在千聚后台申请的API Key
        String apiKey = "你的千聚API_KEY";
        
        OkHttpClient client = new OkHttpClient().newBuilder()
                .connectTimeout(60, TimeUnit.SECONDS)
                .readTimeout(60, TimeUnit.SECONDS)
                .build();
        
        JSONObject requestBody = new JSONObject();
        requestBody.put("model", "gpt-3.5-turbo");
        // 构建消息
        JSONObject message = new JSONObject();
        message.put("role", "user");
        message.put("content", "用中文回答：Java中String和StringBuilder的区别");
        requestBody.put("messages", new JSONObject[]{message});
        
        MediaType mediaType = MediaType.parse("application/json");
        Request request = new Request.Builder()
                .url(baseUrl + "/chat/completions")
                .method("POST", RequestBody.create(mediaType, requestBody.toJSONString()))
                .addHeader("Authorization", "Bearer " + apiKey)
                .addHeader("Content-Type", "application/json")
                .build();
        
        try {
            Response response = client.newCall(request).execute();
            String responseBody = response.body().string();
            System.out.println("结果: " + responseBody);
        } catch (Exception e) {
            e.printStackTrace();
        }
    }
}
