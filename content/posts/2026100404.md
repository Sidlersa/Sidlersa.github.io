python
import base64
import requests

# 你的千聚 API Key
API_KEY = "sk-your-qianjuai-api-key-here"

# 图片文件路径（换成你想分析的图片）
IMAGE_PATH = "your_image.jpg"

# 读取并转为Base64
with open(IMAGE_PATH, "rb") as image_file:
    base64_image = base64.b64encode(image_file.read()).decode("utf-8")

# 调用千聚接口，使用GPT-4o模型分析图片
response = requests.post(
    url="https://www.qianjuai.com/v1/chat/completions",
    headers={
        "Authorization": f"Bearer {API_KEY}",
        "Content-Type": "application/json",
    },
    json={
        "model": "gpt-4o",
        "messages": [
            {
                "role": "user",
                "content": [
                    {
                        "type": "text",
                        "text": "请详细描述这张图片中有什么，并分析其核心信息。"
                    },
                    {
                        "type": "image_url",
                        "image_url": {
                            "url": f"data:image/jpeg;base64,{base64_image}",
                            "detail": "high"
                        }
                    }
                ]
            }
        ],
        "max_tokens": 1000
    }
)

# 输出结果
if response.status_code == 200:
    result_text = response.json()["choices"][0]["message"]["content"]
    print("🤖 图片理解结果：\n")
    print(result_text)
else:
    print("❌ 请求失败：", response.status_code, response.text)


**解释这段代码的作用：**

- 告诉AI要分析一张本地图片；
- 把图片转换成AI能理解的语言（Base64）；
- 调用千聚的API（注意这里用了`www.qianjuai.com/v1`），代理向GPT-4o发出请求；
- 模型的`detail: "high"`表示开启高清晰度模式，图片细节更完整；
- 最终AI返回一段文字，描述图片里的内容，并给出分析。
