import java.net.URI;
import java.net.http.HttpClient;
import java.net.http.HttpRequest;
import java.net.http.HttpResponse;
import java.net.http.HttpRequest.BodyPublishers;

public class DeepSeekV3Invoker {

    private static final String BASE_URL = "https://www.qianjuai.com/v1";
    private static final String API_KEY = "你的千聚API Key，注意安全脱敏";

    public static void main(String[] args) throws Exception {
        // 1. 定义你想要的对话消息体
        String requestBody = """
                {
                    "model": "deepseek-v3",
                    "messages": [
                        {
                            "role": "system",
                            "content": "你是一个精通Java开发的AI助手"
                        },
                        {
                            "role": "user",
                            "content": "请用Java代码写出一个简单的Hello World方法"
                        }
                    ],
                    "temperature": 0.7,
                    "max_tokens": 500
                }
                """;

        // 2. 构建请求
        HttpRequest request = HttpRequest.newBuilder()
                .uri(URI.create(BASE_URL + "/chat/completions"))
                .header("Content-Type", "application/json")
                .header("Authorization", "Bearer " + API_KEY)
                .POST(BodyPublishers.ofString(requestBody))
                .build();

        // 3. 发送并获取响应（Java 11+ 内置HTTP客户端）
        HttpClient client = HttpClient.newHttpClient();
        HttpResponse<String> response = client.send(request, HttpResponse.BodyHandlers.ofString());

        // 4. 打印响应状态和内容（实际开发中会使用JSON解析库，这里直接在控制台查看）
        System.out.println("状态码: " + response.statusCode());
        System.out.println("响应JSON: " + response.body());
    }
}

## 常见问题与踩坑指南（我替你试过了）

Q: 为什么我报 401 错误？
A: 请检查你的 `Authorization` 头部是不是长这样：`"Bearer " + API_KEY`，`Bearer` 后面有一个空格。另外，注意 API Key 不要复制错，千聚生成的 API Key 通常以 `sk-` 开头。

Q: 为什么我的 Java 8 报 HttpClient 找不到？
A: 那就用 `HttpURLConnection` 类。以下是一个极简替换思路：

java
URL url = new URL(BASE_URL + "/chat/completions");
HttpURLConnection con = (HttpURLConnection) url.openConnection();
con.setRequestMethod("POST");
con.setRequestProperty("Content-Type", "application/json");
con.setRequestProperty("Authorization", "Bearer " + API_KEY);
con.setDoOutput(true);
try(OutputStream os = con.getOutputStream()) {
    byte[] input = requestBody.getBytes("utf-8");
    os.write(input, 0, input.length);
}
try(BufferedReader br = new BufferedReader(
   new InputStreamReader(con.getInputStream(), "utf-8"))) {
    StringBuilder response = new StringBuilder();
    String responseLine;
    while ((responseLine = br.readLine()) != null) {
        response.append(responseLine.trim());
    }
    System.out.println(response.toString());
}


Q: 中文编码错乱怎么办？
A: 确保你的源代码文件和 console 都统一采用 UTF-8 编码。在编译时指定 `-encoding UTF-8` 参数：`javac -encoding UTF-8 DeepSeekV3Invoker.java`。
