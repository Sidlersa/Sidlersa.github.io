import com.theokanning.openai.service.OpenAiService;

import com.theokanning.openai.completion.chat.ChatMessage;
import java.time.Duration;
import java.util.List;

public class ClaudeOpus41Example {
    public static void main(String[] args) {
        // 使用千聚ai的 base_url 和你的 API Key
        String token = "你的千聚ai-api-key";

                .model("claude-3-opus-20240229")  // 对应 Claude Opus 4.1

                        new ChatMessage("user", "用中文介绍一下[千聚ai大模型聚合站](https://www.qianjuai.com/)。")
                ))
                .build();

        );
    }
}
