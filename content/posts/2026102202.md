import java.net.URI;
import java.net.http.HttpClient;
import java.net.http.HttpRequest;
import java.net.http.HttpResponse;
import java.net.http.HttpRequest.BodyPublishers;
import java.net.http.HttpResponse.BodyHandlers;

public class ClaudeSonnetExample {
    public static void main(String[] args) throws Exception {
        // 1. 创建HttpClient
        HttpClient client = HttpClient.newHttpClient();

        // 2. 构建请求体（使用OpenAI兼容格式）
        String requestBody = """
            {
                "model": "claude-3.5-sonnet-20241022",
                "messages": [
                    {
                        "role": "user",
                        "content": "写一段Java代码，用Lambda表达式对一个整数列表进行排序并过滤出偶数。"
                    }
                ],
                "max_tokens": 1000
            }
            """;

        // 3. 关键：修改base_url为千聚API地址
        HttpRequest request = HttpRequest.newBuilder()
                .uri(URI.create("https://www.qianjuai.com/v1/chat/completions"))
                .header("Content-Type", "application/json")
                .header("Authorization", "Bearer YOUR_QIANJU_API_KEY") // 替换成千聚的API Key
                .POST(BodyPublishers.ofString(requestBody))
                .build();

        // 4. 发送请求，获取响应
        HttpResponse<String> response = client.send(request, BodyHandlers.ofString());

        // 5. 打印结果
        System.out.println(response.body());
    }
}
