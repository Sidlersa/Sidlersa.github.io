import java.net.URI;
import java.net.http.HttpClient;
import java.net.http.HttpRequest;
import java.net.http.HttpResponse;

public class QianjuAiExample {
    public static void main(String[] args) throws Exception {
        // 1. 创建HttpClient
        HttpClient client = HttpClient.newHttpClient();

        // 2. 构造请求体
        String requestBody = """
            {
              "model": "gpt-5.2pro",
              "messages": [
                {"role": "system", "content": "你是一个Java专家。"},
                {"role": "user", "content": "写一个简单的Spring Boot控制器。"}
              ],
              "max_tokens": 500
            }
            """;

        // 3. 构建请求
        HttpRequest request = HttpRequest.newBuilder()
                .uri(URI.create("https://www.qianjuai.com/v1")) // 关键：替换接口地址
                .header("Content-Type", "application/json")
                .header("Authorization", "Bearer 你的API_KEY") // 关键：替换密钥
                .POST(HttpRequest.BodyPublishers.ofString(requestBody))
                .build();

        // 4. 发送并接收响应
        HttpResponse<String> response = client.send(request, HttpResponse.BodyHandlers.ofString());

        // 5. 输出结果
        System.out.println("状态码: " + response.statusCode());
        System.out.println("回复内容: " + response.body());
    }
}
