import cn.hutool.http.HttpRequest;
import cn.hutool.http.HttpResponse;
import cn.hutool.json.JSONArray;
import cn.hutool.json.JSONObject;

public class GLMClientExample {

    // 1. 关键一步：把 baseUrl 从官方地址改为千聚api地址
    private static final String BASE_URL = "https://www.qianjuai.com/v1";
    // 2. 你的[千聚ai聚合站](https://www.qianjuai.com/) API Key
    private static final String API_KEY = "sk-your-qianjuai-key";

    public static void main(String[] args) {
        JSONObject requestBody = new JSONObject();
        requestBody.setOpt("model", "glm-4-plus"); // 模型名完全兼容
        JSONArray messages = new JSONArray();
        JSONObject userMessage = new JSONObject();
        userMessage.setOpt("role", "user");
        userMessage.setOpt("content", "用中文介绍一下你自己");
        messages.add(userMessage);
        requestBody.setOpt("messages", messages);
        requestBody.setOpt("stream", false);

        try (HttpResponse response = HttpRequest.post(BASE_URL + "/chat/completions")
                .header("Authorization", "Bearer " + API_KEY)
                .header("Content-Type", "application/json")
                .body(requestBody.toString())
                .execute()) {

            if (response.isOk()) {
                JSONObject jsonObject = new JSONObject(response.body());
                System.out.println("[千聚ai聚合站](https://www.qianjuai.com/) GLM回复: " + jsonObject.getJSONArray("choices")
                        .getJSONObject(0)
                        .getJSONObject("message")
                        .getStr("content"));
            } else {
                System.err.println("请求失败: " + response.getStatus() + " - " + response.body());
            }
        }
    }
}
