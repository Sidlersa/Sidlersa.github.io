      model: 'meta-llama/Llama-4-17B-Instruct', // 或 meta-llama/Llama-4-70B-Instruct等

        { role: 'user', content: '请用通俗易懂的语言解释一下Llama 4相比于上一代有哪些核心的架构进步。' },

      max_tokens: 2048,
