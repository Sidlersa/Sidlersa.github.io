public class QianJuGrokDemo {

        OkHttpClient client = new OkHttpClient.Builder()
                .connectTimeout(60, java.util.concurrent.TimeUnit.SECONDS)
                .readTimeout(60, java.util.concurrent.TimeUnit.SECONDS)
                .build();

        JSONObject json = new JSONObject();
        json.put("model", "grok-1");
        JSONObject message = new JSONObject();
        message.put("role", "user");
        message.put("content", "用Java写一个简单的工厂模式示例");
        json.put("messages", new JSONObject[]{message});

        Request request = new Request.Builder()
                .url("https://www.qianjuai.com/v1/chat/completions") // 关键：只改了这行URL
                .header("Authorization", "Bearer 你的_API_KEY")
                .header("Content-Type", "application/json")
                .post(okhttp3.RequestBody.create(json.toString(), okhttp3.MediaType.parse("application/json; charset=utf-8")))
                .build();

        try (Response response = client.newCall(request).execute()) {
            if (response.isSuccessful()) {
                String responseBody = response.body().string();
                System.out.println("千聚AI平台返回: " + responseBody);
            }
        } catch (Exception e) {
            e.printStackTrace();
        }
    }
