import okhttp3.*;
import com.google.gson.JsonArray;
import com.google.gson.JsonObject;

import java.io.IOException;

public class DoubaoChatExample {

    // 你的千聚 API Key（从[千聚ai大模型聚合站](https://www.qianjuai.com/)后台获取）
    private static final String API_KEY = "your_qianju_api_key_here";

    // 豆包模型标识（例如：doubao-pro-32k）
    private static final String MODEL = "doubao-pro-32k";

    // 千聚提供的 API 基地址（兼容 OpenAI 格式）
    private static final String BASE_URL = "https://www.qianjuai.com/v1";

    public static void main(String[] args) throws IOException {
        // 1. 构建请求体
        JsonObject requestBody = new JsonObject();
        requestBody.addProperty("model", MODEL);

        JsonArray messages = new JsonArray();

        // 系统消息（可选，设定对话背景）
        JsonObject systemMessage = new JsonObject();
        systemMessage.addProperty("role", "system");
        systemMessage.addProperty("content", "你是一个专业的AI助手，简称豆包。");
        messages.add(systemMessage);

        // 用户消息
        JsonObject userMessage = new JsonObject();
        userMessage.addProperty("role", "user");
        userMessage.addProperty("content", "请用Java写一个简单的Hello World程序。");
        messages.add(userMessage);

        requestBody.add("messages", messages);
        requestBody.addProperty("temperature", 0.7); // 控制回复随机性
        requestBody.addProperty("max_tokens", 1024); // 最大输出长度

        // 2. 构建 HTTP 请求
        OkHttpClient client = new OkHttpClient().newBuilder()
                .connectTimeout(30, java.util.concurrent.TimeUnit.SECONDS)
                .readTimeout(30, java.util.concurrent.TimeUnit.SECONDS)
                .build();

        MediaType JSON = MediaType.get("application/json; charset=utf-8");
        Request request = new Request.Builder()
                .url(BASE_URL + "/chat/completions")
                .header("Authorization", "Bearer " + API_KEY)
                .header("Content-Type", "application/json")
                .post(RequestBody.create(JSON, requestBody.toString()))
                .build();

        // 3. 发送请求并处理响应
        try (Response response = client.newCall(request).execute()) {
            if (response.isSuccessful() && response.body() != null) {
                String responseBody = response.body().string();
                System.out.println("豆包回复：");
                System.out.println(responseBody);

                // 可选：解析 JSON 提取回复内容

                //         .get("content").getAsString();
                // System.out.println(content);
            } else {
                System.err.println("请求失败（状态码 " + response.code() + "）：" + response.message());
            }
        }
    }
}

4.  **认证**：通过请求头 `Authorization: Bearer <your_api_key>` 进行认证，这个 API key 在千聚后台一键生成。

## 避坑指南：五个最容易踩的雷

### 1. API Key 泄漏
**雷**：把 API Key 硬编码在代码里，推到公共仓库（GitHub）上，等于把家门的密码贴在大街上。
**避坑**：使用环境变量或配置文件（如 `application.properties`）存储 API Key，确保 `.gitignore` 文件忽略这些敏感文件。

### 2. 模型标识写错
**雷**：豆包模型有多个版本（如 `doubao-pro-32k`、`doubao-lite-32k`），写错了调用失败。
**避坑**：从千聚后台或官方文档复制准确的模型名称。可以先在调试工具（如 Postman）里测试一次。

### 3. 忽略调用频率限制
**雷**：写个循环高频调用，超过 API 限额后请求失败，甚至被临时封禁。
**避坑**：实现重试机制和限流逻辑。比如使用 `Retrofit` + `OkHttp` 的拦截器实现指数退避策略。

### 4. 忽略超时设置
**雷**：模型推理时间较长，默认连接超时设置为 10 秒，容易导致“连接超时”错误。
**避坑**：设置合理的超时时间（如 30s 连接超时、60s 读取超时），尤其是处理较复杂提示时。

### 5. 忽略流式输出的特殊性
**雷**：想做大模型“打字机”效果，没有正确处理 SSE 事件流，导致页面空白或乱码。
**避坑**：如果要流式输出，在请求中设置 `"stream": true`，并使用支持 SSE 的客户端（如 `OkHttp` 的 `EventSource`）逐行解析返回数据。
