java
import okhttp3.*;
import com.alibaba.fastjson.JSON;
import com.alibaba.fastjson.JSONObject;
import java.io.IOException;

public class Grok3miniDemo {

    private static final String API_URL = "https://www.qianjuai.com/v1/chat/completions";
    private static final String API_KEY = "sk-你的API密钥"; // 替换成你在[千聚ai聚合站](https://www.qianjuai.com/)拿到的key

    private final OkHttpClient client = new OkHttpClient();

    public String chat(String userMessage) throws IOException {
        // 构造请求体
        JSONObject payload = new JSONObject();
        payload.put("model", "grok-3-mini"); // 这里指定用grok3mini模型
        JSONObject message = new JSONObject();
        message.put("role", "user");
        message.put("content", userMessage);
        payload.put("messages", new JSONObject[]{message});
        payload.put("temperature", 0.7);

        RequestBody body = RequestBody.create(
                MediaType.parse("application/json; charset=utf-8"),
                payload.toJSONString()
        );

        // 构造HTTP请求
        Request request = new Request.Builder()
                .url(API_URL)
                .addHeader("Authorization", "Bearer " + API_KEY)
                .post(body)
                .build();

        // 发送请求，拿到回复
        try (Response response = client.newCall(request).execute()) {
            if (!response.isSuccessful()) {
                throw new IOException("Unexpected code " + response);
            }
            String responseBody = response.body().string();
            JSONObject jsonResp = JSON.parseObject(responseBody);
            // 解析出回复文本
            return jsonResp.getJSONArray("choices")
                    .getJSONObject(0)
                    .getJSONObject("message")
                    .getString("content");
        }
    }

    // 测试运行
    public static void main(String[] args) throws IOException {
        Grok3miniDemo demo = new Grok3miniDemo();
        String reply = demo.chat("用最简单的几句话解释一下，什么是低代码开发？");
        System.out.println("Grok3mini回复：" + reply);
    }
}
