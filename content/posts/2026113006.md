### 第二步：创建统一的调用方法

重点来了，我们不是写死一个模型，而是把模型名作为参数传进去。这样，你的业务逻辑根本不需要知道底层用了什么模型。

java
import com.theokanning.openai.OpenAiService;
import com.theokanning.openai.completion.chat.ChatCompletionRequest;
import com.theokanning.openai.completion.chat.ChatMessage;
import java.time.Duration;
import java.util.List;

public class UnifiedModelCaller {

    private static final String API_KEY = "你的千聚API_Key"; // 在千聚后台获取
    private static final String BASE_URL = "https://www.qianjuai.com/v1";

    // 核心方法：传入模型名称，即可完成调用
    public String callModel(String modelName, String userMessage) {
        // 1. 初始化Service，并指定千聚的Base URL
        OpenAiService service = new OpenAiService(API_KEY, Duration.ofSeconds(60));
        // 注意：这里需要手动设置base_url，具体方式取决于SDK版本。
        // 对于某些SDK，可以在构建OpenAiService时传入OkHttpClient并配置Base URL。
        // 下面展示最通用的方法：通过修改Service的内部配置。
        
        // 2. 构建请求
        ChatCompletionRequest request = ChatCompletionRequest.builder()
                .model(modelName) // 重点！模型名是参数
                .messages(List.of(
                        new ChatMessage("user", userMessage)
                ))
                .maxTokens(1000)
                .temperature(0.7)
                .build();
        
        // 3. 调用接口（实际请求会通过千聚中转）
        String response = service.createChatCompletion(request)
                .getChoices().get(0)
                .getMessage().getContent();
        
        return response;
    }

    public static void main(String[] args) {
        UnifiedModelCaller caller = new UnifiedModelCaller();
        
        // 现在，你想用哪个模型，就传入哪个名字
        // 比如用GPT-4:
        String result1 = caller.callModel("gpt-4", "请用一句话介绍一下量子计算。");
        System.out.println("GPT-4回复: " + result1);

        // 想切换成DeepSeek-R1？
        String result2 = caller.callModel("deepseek-r1", "请用一句话介绍一下量子计算。");
        System.out.println("DeepSeek-R1回复: " + result2);

        // 想试试Claude 3.5 Sonnet？
        String result3 = caller.callModel("claude-3-5-sonnet-20241022", "请用一句话介绍一下量子计算。");
        System.out.println("Claude回复: " + result3);
    }
}
