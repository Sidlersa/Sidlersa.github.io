import com.openai.client.OpenAIClient;
import com.openai.client.okhttp.OpenAIOkHttpClient;
import com.openai.models.ChatCompletionCreateParams;
import com.openai.models.ChatCompletion;

public class QianJuDemo {
    public static void main(String[] args) {
        OpenAIClient client = OpenAIOkHttpClient.builder()
                .baseUrl("https://www.qianjuai.com/v1") // 千聚的 API 地址
                .apiKey("你的千聚API Key") // 在千聚后台获取
                .build();

        ChatCompletionCreateParams params = ChatCompletionCreateParams.builder()
                .model("kimi-k2-thinking") // 指定模型
                .addUserMessage("为我写一个Java的线程池示例")
                .build();

        ChatCompletion completion = client.chat().completions().create(params);
        System.out.println(completion.choices().get(0).message().content().get());
    }
}
